Chart in PyScript

import json import pyodide from js import Bokeh, console, JSON from bokeh.embed import json_item from bokeh.plotting import figure from bokeh.resources import CDN cars = ['Ford F-Series', 'Chevy Silverado', 'Ram-Pickup', 'Toyota RAV4', 'Toyota Camry'] counts = [3063909, 2211745, 2235102, 1686135, 1240322] p = figure(x_range=cars, height=350, title="Car Sales", toolbar_location=None, tools="") p.vbar(x=cars, top=counts, width=0.9) p.xgrid.grid_line_color = None p.y_range.start = 0 p_json = json.dumps(json_item(p, "chart")) Bokeh.embed.embed_item(JSON.parse(p_json))
